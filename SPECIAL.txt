<script>
  const steps = document.querySelectorAll('.step');
  let current = 0;

  function nextStep() {
    gsap.to(steps[current], { opacity: 0, scale: 0.95, duration: 0.4 });
    setTimeout(() => {
      steps[current].classList.remove('active');
      current++;
      steps[current].classList.add('active');
      gsap.fromTo(steps[current], { opacity: 0, y: 40 }, { opacity: 1, y: 0, duration: 0.6 });
      document.getElementById('progress').style.width = ((current) / (steps.length - 1)) * 100 + '%';
    }, 400);
  }

  function celebrate() {
    const btn = document.getElementById('celebrateBtn');
    btn.disabled = true;
    gsap.to(btn, { opacity: 0, duration: 0.5 });

    gsap.to('#finalText', { opacity: 1, y: -10, duration: 1, delay: 0.5 });

    const duration = 5000;
    const end = Date.now() + duration;

    (function frame() {
      confetti({
        particleCount: 6,
        spread: 70,
        origin: { x: Math.random(), y: 1 }
      });
      if (Date.now() < end) requestAnimationFrame(frame);
    })();
  }

  /* THREE.JS HEARTS */
  const scene = new THREE.Scene();
  const camera = new THREE.PerspectiveCamera(60, window.innerWidth/window.innerHeight, 0.1, 1000);
  camera.position.z = 30;

  const renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('three-bg'), alpha: true });
  renderer.setSize(window.innerWidth, window.innerHeight);

  const hearts = [];
  const geometry = new THREE.TorusKnotGeometry(0.6, 0.2, 100, 16);

  for (let i = 0; i < 25; i++) {
    const material = new THREE.MeshStandardMaterial({
      color: 0xff4d6d,
      metalness: 0.6,
      roughness: 0.3
    });
    const mesh = new THREE.Mesh(geometry, material);
    mesh.position.set(
      (Math.random() - 0.5) * 40,
      (Math.random() - 0.5) * 40,
      (Math.random() - 0.5) * 40
    );
    scene.add(mesh);
    hearts.push(mesh);
  }

  const light = new THREE.PointLight(0xffffff, 1);
  light.position.set(10, 10, 20);
  scene.add(light);

  function animate() {
    requestAnimationFrame(animate);
    hearts.forEach(h => {
      h.rotation.x += 0.002;
      h.rotation.y += 0.003;
      h.position.y += Math.sin(Date.now() * 0.001) * 0.002;
    });
    renderer.render(scene, camera);
  }
  animate();
</script>
